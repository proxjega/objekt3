184float Student::calculateFinalMean() const {
185 vector<int> marks = getMarks();
186 if (marks.size() == 0) return 0;
187 return 0.4 * ((std::accumulate(marks.begin(), marks.end(), 0) * 1.0) / (marks.size() * 1.0)) + 0.6 * examMark_;

193float Student::calculateFinalMedian() const {
